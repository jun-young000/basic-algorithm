### 함수 선언부(클래스 선언부)
def add_data(friend):
    katok.append(None)
    kLen = len(katok)
    katok[kLen - 1] = friend

def insert_data(position,friend):
    katok.append(None)
    klen=len(katok)
    for i in range(klen-1,position, -1):
        katok[i]=katok[i-1]
        katok[i-1]=None

    katok[position]=friend

def delete_data(position):
    klen=len(katok)
    katok[position]=None

    for i in range(position+1,klen,1):
        katok[i-1]=katok[i]
        katok[i]=None

    del(katok[klen-1])
